Run regression model: Math (Y) regressed on Computer (X)

reg_model <- lm(formula = math ~ computer,
   data = ca_schools)


summ(reg_model, digits = 3, model.fit = FALSE)

Plot the linear regression model (line) & the data (points): Math (Y) regressed on Computer (X)

plot_reg(
  data = ca_schools,
  x_var = computer,
  x_label = "Number of Computers",
  y_var = math,
  y_label = "Math Scores",
  point_color = "lightblue",
  reg_line = TRUE, eq_label = TRUE, digits = 3,
  x_min = 0, x_max = 3324, x_by = 500,
  y_min = 605, y_max = 710, y_by = 100,
  title   = ""
)

Data source:

Panel Study on Income Dynamics (https://www.rdocumentation.org/packages/AER/versions/1.2-9/topics/PSID1982)

set.seed(54321)

income_data <- read_csv("https://raw.githubusercontent.com/ejvanholm/DataProjects/master/PSID1982.csv") %>% 
  sample_n(500)

Let’s take a look at the distribution of our outcome variable wage

hist_plot(
  data = income_data,
  x_var = wage,
  x_label = "Income",
  binwidth=100,
  fill = "green3",
  color = "black",
  #break_min = 0, break_max = 35, break_by = 10,
  mean_line = TRUE, median_line = TRUE,
  skew_value = TRUE, kurtosis_value = TRUE,
  title = "Distribution of Income in Adult Sample (N=500)")

Estimate the regression model using lm()

model_wage <- lm(wage ~ education, income_data)

summ(model_wage, digits = 3, model.fit = FALSE)
plot_reg(
  data = income_data,
  x_var = education,
  x_label = "Years of Education",
  y_var = wage,
  y_label = "Income (Wage)",
  point_color = "green3",
  reg_line = TRUE,
  eq_label = TRUE, digits = 0,
  x_min = 3, x_max = 18, x_by = 1,
  y_min = 290, y_max = 5100, y_by = 500,
  title   = "Does Years of Education Predict Income?")
